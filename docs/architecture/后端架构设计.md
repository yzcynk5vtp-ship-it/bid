# 西域智慧供应链投标管理平台 - 后端架构设计

> 基于Backend Patterns最佳实践的生产级后端系统

---

## 一、技术栈选型

```
语言:        Node.js + TypeScript
框架:        NestJS (企业级框架)
数据库:      MySQL 8.0 (主库) + Redis (缓存)
ORM:         Prisma
认证:        JWT + Passport
消息队列:    BullMQ (基于Redis)
文档:       Swagger/OpenAPI
部署:       Docker + Kubernetes
```

### 2.2 表结构设计 (Prisma Schema)

```prisma
// ==================== 基础模块 ====================

model User {
  id            String    @id @default(cuid())
  username      String    @unique
  email         String    @unique
  phone         String?
  name          String
  avatar        String?
  role          UserRole  @default(STAFF)
  departmentId  String?
  department    Department? @relation(fields: [departmentId], references: [id])
  status        UserStatus @default(ACTIVE)
  lastLoginAt   DateTime?
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt

  // 关系
  createdTenders Tender[] @relation("Creator")
  assignedTasks Task[]    @relation("Assignee")
  createdTasks  Task[]    @relation("Creator")
  approvals     Approval[]
  auditLogs     AuditLog[]
  accounts      Account[]
  expenses      Expense[]

  @@index([departmentId])
  @@index([role])
  @@index([status])
}

enum UserRole {
  ADMIN       // 系统管理员
  MANAGER     // 部门经理
  SALES       // 销售
  BIDDING     // 投标经理
  TECHNICAL   // 技术人员
  FINANCE     // 财务
  STAFF       // 普通员工
}

enum UserStatus {
  ACTIVE
  INACTIVE
  LOCKED
  PENDING
}

model Department {
  id        String   @id @default(cuid())
  name      String   @unique
  code      String   @unique
  parentId  String?
  parent    Department? @relation("Hierarchy", fields: [parentId], references: [id])
  children  Department[] @relation("Hierarchy")
  sort      Int      @default(0)
  status    DeptStatus @default(ACTIVE)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  // 关系
  users User[]

  @@index([parentId])
  @@index([code])
}

enum DeptStatus {
  ACTIVE
  INACTIVE
}

// ==================== 标讯模块 ====================

model Tender {
  id                String          @id @default(cuid())
  title             String
  source            TenderSource
  sourceUrl         String?
  projectCode       String?
  budget            Decimal?        @db.Decimal(15, 2)
  region            String?
  industry          String?
  purchaser         String          // 采购方
  purchaserType     String?
  deadline          DateTime?
  openingDate       DateTime?
  publishDate       DateTime?
  description       String?         @db.Text
  requirements       String?         @db.Text
  aiScore           Int?            // AI评分
  winProbability    String?         // 中标概率: high/medium/low
  recommendation    String?         @db.Text // AI推荐理由
  status            TenderStatus    @default(NEW)
  creatorId         String
  creator           User            @relation("Creator")
  assignedTo        String?
  assignee          User?           @relation("Assignee", fields: [assignedTo], references: [id])
  followUpStatus    FollowUpStatus? @default(NULL)
  project           Project?
  createdAt         DateTime        @default(now())
  updatedAt         DateTime        @updatedAt

  // 关系
  analysis          TenderAnalysis?

  @@index([status])
  @@index([assignedTo])
  @@index([followUpStatus])
  @@index([deadline])
  @@index([aiScore])
}

enum TenderSource {
  INTERNAL    // 内部
  EXTERNAL    // 外部获取
  MANUAL      // 人工录入
}

enum TenderStatus {
  NEW           // 新建
  CONTACTED     // 已联系
  FOLLOWING     // 跟进中
  QUOTING       // 报价中
  BIDDING       // 投标中
  ABANDONED     // 已放弃
  CONVERTED     // 已转项目
}

enum FollowUpStatus {
  NULL
  CONTACTED
  IN_PROGRESS
  PROPOSAL_SENT
  NEGOTIATING
}

model TenderAnalysis {
  id                  String   @id @default(cuid())
  tenderId            String   @unique
  tender              Tender   @relation(fields: [tenderId], references: [id])
  // AI分析结果
  customerHistory      Json?    // 客户历史分析
  requirementAnalysis  Json?    // 需求分析
  competitorAnalysis   Json?    // 竞争对手分析
  riskWarning         Json?    // 风险预警
  bidStrategy         Json?    // 投标策略建议
  analyzedAt          DateTime @default(now())
}

// ==================== 项目模块 ====================

model Project {
  id                  String            @id @default(cuid())
  code                String            @unique // 项目编号
  name                String
  tenderId            String?           @unique
  tender              Tender?           @relation(fields: [tenderId], references: [id])
  customerId          String?
  customerName        String
  budget              Decimal?          @db.Decimal(15, 2)
  actualAmount        Decimal?          @db.Decimal(15, 2)
  region              String?
  industry            String?
  managerId           String?
  manager             User?             @relation("Manager", fields: [managerId], references: [id])
  startDate           DateTime?
  endDate             DateTime?
  deadline            DateTime?
  status              ProjectStatus     @default(DRAFT)
  progress            Int               @default(0) // 0-100

  // 投标结果
  bidResult           BidResult?
  contractAmount      Decimal?          @db.Decimal(15, 2)
  contractStartDate   DateTime?
  contractEndDate     DateTime?

  description         String?           @db.Text
  createdAt           DateTime          @default(now())
  updatedAt           DateTime          @updatedAt

  // 关系
  tasks               Task[]
  bidProcess          BidProcess?
  competitors         ProjectCompetitor[]

  @@index([status])
  @@index([managerId])
  @@index([deadline])
}

enum ProjectStatus {
  DRAFT        // 草稿
  PENDING      // 待审批
  APPROVED     // 已审批
  IN_PROGRESS  // 进行中
  REVIEW       // 审核中
  SUBMITTED    // 已提交
  WON          // 中标
  LOST         // 未中标
  CANCELLED    // 已取消
}

enum BidResult {
  WON          // 中标
  LOST         // 未中标
  ABANDONED    // 放弃
  PENDING      // 待定
}

model ProjectCompetitor {
  id              String      @id @default(cuid())
  projectId       String
  project         Project     @relation(fields: [projectId], references: [id])
  competitorName  String
  competitorType  String?
  price           Decimal?    @db.Decimal(15, 2)
  discount        String?
  paymentTerm     String?
  strength        String?     @db.Text
  weakness        String?     @db.Text
  strategy        String?     @db.Text
  createdAt       DateTime    @default(now())
  updatedAt       DateTime    @updatedAt

  @@index([projectId])
}

model Task {
  id              String      @id @default(cuid())
  projectId       String
  project         Project     @relation(fields: [projectId], references: [id], onDelete: Cascade)
  title           String
  description     String?     @db.Text
  type            TaskType
  priority        TaskPriority
  status          TaskStatus  @default(TODO)
  assigneeId      String?
  assignee        User?       @relation("Assignee", fields: [assigneeId], references: [id])
  creatorId       String
  creator         User        @relation("Creator", fields: [creatorId], references: [id])
  dueDate         DateTime?
  estimatedHours  Decimal?
  actualHours     Decimal?
  parentTaskId    String?
  sortOrder       Int         @default(0)
  createdAt       DateTime    @default(now())
  updatedAt       DateTime    @updatedAt

  // 关系
  deliverables    Deliverable[]

  @@index([projectId])
  @@index([assigneeId])
  @@index([status])
  @@index([dueDate])
}

enum TaskType {
  REQUIREMENT     // 需求分析
  TECHNICAL      // 技术方案
  COMMERCIAL     // 商务应答
  QUALIFICATION  // 资质准备
  PRICING        // 报价
  REVIEW         // 审核
  SUBMISSION     // 提交
}

enum TaskPriority {
  URGENT
  HIGH
  MEDIUM
  LOW
}

enum TaskStatus {
  TODO           // 待办
  IN_PROGRESS    // 进行中
  REVIEW         // 待审核
  COMPLETED      // 已完成
  CANCELLED      // 已取消
}

model Deliverable {
  id              String      @id @default(cuid())
  taskId          String
  task            Task        @relation(fields: [taskId], references: [id], onDelete: Cascade)
  title           String
  fileName        String
  fileUrl         String
  fileSize        Int?
  version         Int         @default(1)
  uploadedBy      String
  uploadedAt      DateTime    @default(now())

  @@index([taskId])
}

model BidProcess {
  id              String      @id @default(cuid())
  projectId       String      @unique
  project         Project     @relation(fields: [projectId], references: [id])

  // 流程步骤
  currentStep     Int         @default(0)

  // 初稿编制
  draftCompleted  Boolean     @default(false)
  draftCompletedAt DateTime?

  // 内部评审
  reviewCompleted Boolean     @default(false)
  reviewCompletedAt DateTime?
  reviewComments  String?     @db.Text

  // 用印申请
  sealCompleted   Boolean     @default(false)
  sealCompletedAt DateTime?
  sealType        String?
  sealReason      String?
  sealFiles       Json?

  // 封装提交
  submitCompleted Boolean     @default(false)
  submitCompletedAt DateTime?
  packageMethod   String?
  packageAddress  String?
  trackingNumber  String?

  createdAt       DateTime    @default(now())
  updatedAt       DateTime    @updatedAt
}
```

```typescript
import { TenderRepository } from './tender.repository'

    private prisma: PrismaService,
    private tenderRepo: TenderRepository,
    private aiService: AIAnalysisService,

    const { page = 1, limit = 20, status, region, industry, aiScore } = query

    const result = await this.prisma.tender.findMany({
      where: {
        ...(status && { status }),
        ...(region && { region }),
        ...(industry && { industry }),
        ...(aiScore && { aiScore: { gte: aiScore } }),
      },
      include: {
        creator: { select: { id: true, name: true, avatar: true } },
        assignee: { select: { id: true, name: true, avatar: true } },
      },
      orderBy: { createdAt: 'desc' },
      skip: (page - 1) * limit,
      take: limit,
    })

    const total = await this.prisma.tender.count({
      where: query,
    })

    return {
      success: true,
      data: result,
      meta: {
        page: Number(page),
        limit: Number(limit),
        total,
        totalPages: Math.ceil(total / limit),
      },
    }
  }

  async findOne(id: string) {

      where: { id },
      include: {
        creator: { select: { id: true, name: true, avatar: true } },
        assignee: { select: { id: true, name: true, avatar: true } },
        analysis: true,
        project: true,
      },
    })

    }

    return {
      success: true,
      data: tender,
    }
  }

      where: { id },

        aiScore: analysis.aiScore,
```

```typescript
@Injectable()
```

```typescript
  constructor(
    private prisma: PrismaService,
    private notificationService: NotificationService,
```

```typescript
  // 每天早上8点执行预警检查
  @Cron(CronExpression.EVERY_DAY_AT_8AM)

    await this.checkQualificationExpiry()
    await this.checkDepositRefund()
    await this.checkProjectDeadline()

  private async checkQualificationExpiry() {
    // 查找30天内到期的资质
    const expiring = await this.prisma.qualification.findMany({
      where: {
        status: 'ACTIVE',
        validUntil: {
          lte: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000),
        },
      },
      include: { createdBy: true },
    })

    for (const qual of expiring) {
      const daysLeft = Math.ceil(
        (qual.validUntil.getTime() - Date.now()) / (1000 * 60 * 60 * 24)
      )

      // 发送预警
      await this.notificationService.sendToRole('ADMIN', {
        type: 'WARNING',
        title: '资质即将到期',
        content: `资质 ${qual.name} 将在 ${daysLeft} 天后到期`,
        data: { qualificationId: qual.id, daysLeft },
      })

      // 标记已提醒
      await this.prisma.qualification.update({
        where: { id: qual.id },
        data: { lastWarnedAt: new Date() },
      })
    }
  }

  private async checkDepositRefund() {
    // 查找7天内到期的保证金
    const deposits = await this.prisma.expense.findMany({
      where: {
        isDeposit: true,
        depositRefunded: false,
        refundReminderSent: false,
      },
      include: { project: true },
    })

    for (const deposit of deposits) {
      // 发送提醒
      await this.notificationService.send({
        userId: deposit.applicantId,
        type: 'REMINDER',
        title: '保证金即将到期',
        content: `项目 ${deposit.projectName} 的保证金需要跟进收回`,
        data: { expenseId: deposit.id },
      })

      // 标记已提醒
      await this.prisma.expense.update({
        where: { id: deposit.id },
        data: { refundReminderSent: true },
      })
    }
  }
```

```typescript
import { Injectable, CanActivate, ExecutionContext } from '@nestjs/common'
import { Reflector } from '@nestjs/core'

@Injectable()
export class PermissionGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    const requiredPermissions = this.reflector.get<string[]>(
      'permissions',
      context.getHandler()
    )

    if (!requiredPermissions) {
      return true
    }

    const request = context.switchToHttp().getRequest()
    const user = request.user

    return this.checkPermissions(user, requiredPermissions)
  }

    return permissions.every(p => rolePermissions.includes(p))
  }
}
```

```typescript
// src/common/utils/encryption.util.ts
import * as crypto from 'crypto'

const ALGORITHM = 'aes-256-gcm'
const KEY = Buffer.from(process.env.ENCRYPTION_KEY, 'hex')

export class EncryptionUtil {
  static encrypt(text: string): string {
    const iv = crypto.randomBytes(16)
    const cipher = crypto.createCipheriv(ALGORITHM, KEY, iv)

    let encrypted = cipher.update(text, 'utf8', 'hex')
    encrypted += cipher.final('hex')

    const authTag = cipher.getAuthTag()

    return `${iv.toString('hex')}:${authTag.toString('hex')}:${encrypted}`
  }

  static decrypt(encryptedText: string): string {
    const [ivHex, authTagHex, encrypted] = encryptedText.split(':')

    const iv = Buffer.from(ivHex, 'hex')
    const authTag = Buffer.from(authTagHex, 'hex')

    const decipher = crypto.createDecipheriv(ALGORITHM, KEY, iv)
    decipher.setAuthTag(authTag)

    let decrypted = decipher.update(encrypted, 'hex', 'utf8')
    decrypted += decipher.final('utf8')

    return decrypted
  }
}
```

```typescript
// src/common/guards/throttler.guard.ts
import { Injectable, NestInterceptor, ExecutionContext } from '@nestjs/common'
import { Observable } from 'rxjs'
import { RedisService } from '../redis/redis.service'

@Injectable()
export class ThrottlerGuard {
  constructor(private redis: RedisService) {}

  async intercept(context: ExecutionContext): Promise<boolean> {
    const request = context.switchToHttp().getRequest()
    const key = `throttle:${request.ip}:${request.route.path}`

    const count = await this.redis.incr(key)

    if (count === 1) {
      await this.redis.expire(key, 60) // 1分钟窗口
    }

    if (count > 100) {
      // 每分钟100次请求
      throw new ThrottlerException('Too many requests')
    }

    return true
  }
}
```
